from abc import abstractmethod, ABC



[docs]class Operator(ABC):
    """Base operator class"""

    @property
    @abstractmethod
    def name(self) -> str:
        """
        Name of the operator
        """

    @property
    @abstractmethod
    def output_args(self) -> list[str]:
        """
        Output arguments of the operator
        """

    @property
    @abstractmethod
    def attributes(self):
        """
        Attributes of the operator
        """
